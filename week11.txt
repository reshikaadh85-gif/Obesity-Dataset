
// FILE 1: Product.cs (Model)

using SQLite;
using System.ComponentModel.DataAnnotations;

public class Product
{
    [PrimaryKey, AutoIncrement]
    public int Id { get; set; }

    [Required(ErrorMessage = "Product name is required")]
    [StringLength(50, MinimumLength = 2, ErrorMessage = "Name must be between 2 and 50 characters")]
    public string Name { get; set; } = string.Empty;

    [Required(ErrorMessage = "Price is required")]
    [Range(0.01, 10000, ErrorMessage = "Price must be between 0.01 and 10000")]
    public decimal Price { get; set; }

    [Required(ErrorMessage = "Stock quantity is required")]
    [Range(0, 1000, ErrorMessage = "Stock must be between 0 and 1000")]
    public int Stock { get; set; }
}



// FILE 2: AppDatabase.cs (Database Connection)

using SQLite;

public class AppDatabase
{
    private SQLiteAsyncConnection _database;

    public AppDatabase()
    {
        // Step 1: Get the database path using FileSystem.AppDataDirectory
        var dbPath = Path.Combine(FileSystem.AppDataDirectory, "products.db3");

        // Step 2: Create SQLite connection
        _database = new SQLiteAsyncConnection(dbPath);

        // Step 3: Create Product table if it doesn't exist
        _database.CreateTableAsync<Product>().Wait();
    }

    public SQLiteAsyncConnection GetConnection()
    {
        return _database;
    }
}



// FILE 3: ProductService.cs (Service Layer)

using SQLite;

public class ProductService
{
    private readonly AppDatabase _appDatabase;
    private SQLiteAsyncConnection _database;

    public ProductService(AppDatabase appDatabase)
    {
        _appDatabase = appDatabase;
        _database = _appDatabase.GetConnection();
    }

    // CREATE - Insert a new product
    public async Task AddProductAsync(Product product)
    {
        await Task.Delay(300); // Simulate delay
        await _database.InsertAsync(product);
    }

    // READ - Get all products
    public async Task<List<Product>> GetAllProductsAsync()
    {
        await Task.Delay(500); // Simulate delay
        return await _database.Table<Product>().ToListAsync();
    }

    // READ - Get product by ID
    public async Task<Product?> GetProductByIdAsync(int id)
    {
        return await _database.Table<Product>()
                              .Where(p => p.Id == id)
                              .FirstOrDefaultAsync();
    }

    // UPDATE - Update existing product
    public async Task UpdateProductAsync(Product product)
    {
        await Task.Delay(300); // Simulate delay
        await _database.UpdateAsync(product);
    }

    // DELETE - Delete a product
    public async Task DeleteProductAsync(int id)
    {
        await Task.Delay(300); // Simulate delay
        var product = await _database.Table<Product>()
                                     .Where(p => p.Id == id)
                                     .FirstOrDefaultAsync();
        if (product != null)
        {
            await _database.DeleteAsync(product);
        }
    }

    // AGGREGATE - Get total product count
    public async Task<int> GetProductCountAsync()
    {
        return await _database.Table<Product>().CountAsync();
    }

    // AGGREGATE - Get total inventory value
    public async Task<decimal> GetTotalInventoryValueAsync()
    {
        var products = await _database.Table<Product>().ToListAsync();
        return products.Sum(p => p.Price * p.Stock);
    }
}

@code {
    private List<Product> products = new List<Product>();
    private Product newProduct = new Product();
    private Product? editingProduct = null;
    private bool isLoading = false;
    private int productCount = 0;
    private decimal totalValue = 0;

    protected override async Task OnInitializedAsync()
    {
        await LoadProducts();
        await LoadStatistics();
    }

    private async Task LoadProducts()
    {
        isLoading = true;
        StateHasChanged();
        products = await ProductService.GetAllProductsAsync();
        isLoading = false;
        StateHasChanged();
    }

    private async Task LoadStatistics()
    {
        productCount = await ProductService.GetProductCountAsync();
        totalValue = await ProductService.GetTotalInventoryValueAsync();
        StateHasChanged();
    }

    private async Task HandleAddProduct()
    {
        isLoading = true;
        await ProductService.AddProductAsync(newProduct);
        newProduct = new Product();
        await LoadProducts();
        await LoadStatistics();
        isLoading = false;
    }

    private void StartEdit(Product product)
    {
        editingProduct = new Product
        {
            Id = product.Id,
            Name = product.Name,
            Price = product.Price,
            Stock = product.Stock
        };
    }

    private async Task HandleUpdateProduct()
    {
        if (editingProduct != null)
        {
            await ProductService.UpdateProductAsync(editingProduct);
            editingProduct = null;
            await LoadProducts();
            await LoadStatistics();
        }
    }

    private void CancelEdit()
    {
        editingProduct = null;
    }

    private async Task HandleDeleteProduct(int id)
    {
        isLoading = true;
        await ProductService.DeleteProductAsync(id);
        await LoadProducts();
        await LoadStatistics();
        isLoading = false;
    }
}

<!-- 
KEY CONCEPTS:

1. SQLITE DATABASE:
   - Local relational database
   - Persistent storage (survives app restart)
   - Structured data with tables and rows
   - Cross-platform (Windows, Android, iOS, Mac)

2. FILESYSTEM:
   - FileSystem.AppDataDirectory: Safe, app-specific location
   - Path.Combine: Cross-platform path creation
   - Database file: products.db3

3. DATABASE CONNECTION:
   - SQLiteAsyncConnection: Async database operations
   - CreateTableAsync<T>: Automatically creates table from class
   - Connection reused throughout app (Singleton pattern)

4. SQLITE ATTRIBUTES:
   - [PrimaryKey]: Defines primary key column
   - [AutoIncrement]: Auto-generates unique IDs
   - Converts C# class to database table automatically

5. CRUD OPERATIONS:
   - CREATE: InsertAsync(object) - Adds new row
   - READ: Table<T>().ToListAsync() - Gets all rows
   - UPDATE: UpdateAsync(object) - Updates existing row
   - DELETE: DeleteAsync(object) - Removes row

6. AGGREGATE FUNCTIONS:
   - CountAsync(): Returns total number of records
   - Custom aggregates using LINQ (Sum, Average, etc.)

7. ARCHITECTURE BENEFITS:
   - Same service layer pattern from Week 10
   - Easy migration from List to SQLite
   - Only changed ProductService implementation
   - UI code unchanged (loose coupling)

8. DEPENDENCY INJECTION:
   - AppDatabase registered as Singleton
   - ProductService registered as Singleton
   - Both injected where needed




NUGET PACKAGE REQUIRED:
Install-Package: sqlite-net-pcl
